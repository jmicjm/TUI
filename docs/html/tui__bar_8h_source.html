132

139  int bar_offset = val_str.size() * !m_display_labels_at_end;
140  int str_offset = getBarSize() * m_display_labels_at_end;
141 
142

144  int h = round(fabs(m_min - m_value) / distance * halves);
145 
146  int dir_c = (direction == tui::DIRECTION::HORIZONTAL ? 0 : getBarSize() - 1);
147 
148  for (int i = 0; i < halves; i += 2)
149  {
150  if (i < h && i + 1 < h)

153  }
154  else if (i < h && i + 1 >= h)

156  surface1D<direction>::setSymbolAt(half, abs(dir_c - i / 2) + bar_offset);
157  }

161  }

169  }
